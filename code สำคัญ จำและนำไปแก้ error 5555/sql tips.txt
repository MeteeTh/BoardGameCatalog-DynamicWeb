UPDATE ตาราง
SET ฟิลด์ = REPLACE(ฟิลด์, ' ', '')
WHERE ฟิลด์ LIKE '% %';
ความหมายคือ REPLACE(ฟิลด์, 'สิ่งที่จะแทน', 'สิ่งที่เป็น')
% หมายถึงการแทนที่สตริงใดๆ ที่มีความยาวได้ที่จะเป็นศูนย์ หรือไม่มีเลยก็ได้ ในส่วนที่ระบุ
' ' หมายถึงสตริงในส่วนที่ระบุ
ตัวอย่างเช่นถ้าเรามีสตริง "hello world" แล้วเงื่อนไข LIKE '% %' จะเลือกข้อมูลนี้เนื่องจากมีช่องว่างอยู่ในสตริง แต่ถ้าเรามีสตริง "helloworld" ก็จะไม่ถูกเลือกเพราะไม่มีช่องว่างอยู่ในสตริง หรือถ้าเป็นคำ'%Children%' ก็จะเลือกข้อมูลที่มี Chilldren อยู่ในสตริงนั้นๆ


ดูฟิลด์ time ว่ารายการไหน มีตัวเลขข้างหน้าต่ำที่สุด เช่น 10-15 ตัวเลขข้างหน้าคือ 10 ไม่นับ - และ null
SELECT *
FROM gamelist
WHERE time IS NOT NULL AND time NOT LIKE '-%' 
-- ไม่รวมค่า NULL และรายการที่ขึ้นต้นด้วย '-'
ORDER BY CAST(CASE 
                WHEN time LIKE '%-%' THEN SUBSTRING_INDEX(time, '-', 1) 
                ELSE time 
             END AS UNSIGNED)
LIMIT 1;
